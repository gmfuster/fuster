null coalescing operator

using System;

class Program {
 static void Main(string[] args) {
    string theString = "hello";
    string theOtherString = null;
    Console.WriteLine(theString ?? "Just Me"); //hello
    Console.WriteLine(theOtherString ?? "Just Me");//Just Me
}
}}
